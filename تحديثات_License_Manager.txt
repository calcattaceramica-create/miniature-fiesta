   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2️⃣ توقيف الترخيص (Suspend License)
   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
   ✓ زر محدث في القائمة: "⏸️ Suspend License"
   ✓ واجهة كاملة لتوقيف التراخيص
   ✓ تحذير قبل التوقيف
   ✓ تأكيد مزدوج
   ✓ تحديث حالة الترخيص في قاعدة البيانات
   ✓ تسجيل تاريخ التوقيف

3️⃣ تفعيل الترخيص (Activate License)
   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
   ✓ زر جديد في القائمة: "▶️ Activate License"
   ✓ واجهة لإعادة تفعيل التراخيص الموقوفة
   ✓ إزالة حالة التوقيف
   ✓ مسح تاريخ التوقيف
   ✓ رسالة نجاح

4️⃣ حذف الترخيص (Delete License)
   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   ✓ واجهة لحذف التراخيص
   ✓ تحذير خطر (Danger Warning)
   ✓ تأكيد مزدوج
   ✓ فحص المستخدمين المرتبطين
   ✓ حذف ناعم (Soft Delete)
   ✓ حماية من الحذف إذا كان هناك مستخدمين

📖 كيفية الاستخدام:
═══════════════════════════════════════════════════════════════════════════════

1. تشغيل License Manager:
   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
   • انقر نقراً مزدوجاً على: Launch_License_Manager.bat
   • أو من CMD: python license_gui.py

2. عرض تفاصيل ترخيص:
   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
   • اضغط على "👁️ License Details"
   • أدخل License Key
   • اضغط "View Details"
   • ستظهر نافذة بجميع التفاصيل

3. توقيف ترخيص:
   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
   • اضغط على "⏸️ Suspend License"
   • أدخل License Key
   • اضغط "Suspend"
   • أكد العملية

4. تفعيل ترخيص:
   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
   • اضغط على "▶️ Activate License"
   • أدخل License Key
   • اضغط "Activate"

5. حذف ترخيص:
   ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
   • اضغط على "🗑️ Delete License"
   • أدخل License Key
   • اضغط "Delete"
   • أكد مرتين (Double Confirmation)
